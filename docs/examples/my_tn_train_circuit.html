    data=U_dense.reshape([2] * (2 * n)),
    inds=[f'k{i}' for i in range(n)] + [f'b{i}' for i in range(n)],
    tags={'U_TARGET'}
)

    return - abs((mpo.H & U).contract(all, optimize='auto-hq')) / np.sqrt(2**n) #(mpo.H @ u_target)  # minus so as to minimize
